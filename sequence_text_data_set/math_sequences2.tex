\text{Let} \ \mathbf{\{a_{n}\}} \ \text{be a sequence and let} \ \mathbf{L} \ \text{be a real number. Given any} \ 
\mathbf{\varepsilon > 0} , \ \text{if an} \ \mathbf{m} \ \text{can be found such that} \ \mathbf{|a_{n}-L| < \varepsilon} \ 
\text{for all} \ \mathbf{n > m},\\ \text{then we say the \textbf{limit} of} \ \mathbf{\{a_{n}\}}, \ \text{as} \ \mathbf{n} \ 
\text{\textbf{approaches infinity}, is} \ \mathbf{L}, \ \text{denoted} 
\ \begin{equation}\lim_{n\to\infty}a_{n} = \mathbf{L}\end{equation}.
